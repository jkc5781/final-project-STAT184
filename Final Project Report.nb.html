Joined World Happiness (.x is 2015 and .y is 2016)

WorldHappiness2015 <-
  WorldHappiness2015 %>%
  mutate(year = "Happiness2015")
WorldHappiness2016 <-
  WorldHappiness2016 %>%
  mutate(year = "Happiness2016")
WorldHappiness <-
  WorldHappiness2015 %>%
  full_join(WorldHappiness2016)
Joining, by = c("Country", "Region", "Happiness.Rank", "Happiness.Score", "Economy..GDP.per.Capita.", "Family", "Health..Life.Expectancy.", "Freedom", "Trust..Government.Corruption.", "Generosity", "Dystopia.Residual", "year")
Column `Country` joining factors with different levels, coercing to character vector
WorldHappiness
RecentSuicideRates <-
  SuicideRates %>%
  filter(year == 2015 | year == 2016)
view(RecentSuicideRates)
RecentSuicideRates <-
  RecentSuicideRates %>%
  select(country, year, sex, population, suicides_no, suicides.100k.pop)

RecentSuicideRates <-
  RecentSuicideRates %>%
  group_by(country, year, sex) %>%
  summarise(totalsuicides = sum(suicides_no))
RecentSuicideRates
WorldHappiness <-
  WorldHappiness %>%
  select(country = Country, Happiness.Score, year)
WorldHappiness
SuicideByYear <-
  RecentSuicideRates %>%
  spread(key = year, value = totalsuicides)
SuicideByYear <-
  SuicideByYear %>%
  na.omit()

names(SuicideByYear)[3] <- "Year2015"
names(SuicideByYear)[4] <- "Year2016"
SuicideByYear
WorldHappinessByYear <-
  WorldHappiness %>%
  spread(key = year, value = Happiness.Score) %>%
  na.omit
WorldHappinessByYear
SuicideHappiness <-
  SuicideByYear %>%
  inner_join(WorldHappinessByYear)
Joining, by = "country"
Column `country` joining factor and character vector, coercing into character vector
SuicideHappiness
